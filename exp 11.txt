CREATE OR REPLACE FUNCTION square1 (n IN NUMBER)
RETURN NUMBER
AS
    square NUMBER;
BEGIN
    square := n * n;
    DBMS_OUTPUT.PUT_LINE('Square of ' || n || ' = ' || square);
    RETURN square;
END;
/
--------------------OUTPUT----------------------
EXEC DBMS_OUTPUT.PUT_LINE(Square1(4));
